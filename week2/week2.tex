		(Filippov 221) Construct the approximations $y_0, y_1, y_2$ for the initial value problem. 
		\begin{parts}
			\part $y' = x - y^2$, $y(0) = 0$;
			\part $y' = y^2 + 3x^2 - 1$, $y(1) = 1$;
			\part $y' = y + e^{y - 1}$, $y(0) = 1$;
			\part $y' = 1 + x \sin y$, $y(\pi) = 2\pi$
		\end{parts}
		
		\question
		(Filippov 223) Indicate any interval of an existence of unique solutions of the initial value problem.
		\begin{parts}
			\part $y' = x + y^3$, $y(0) = 0$;
			\part $y' = 2y^2 - x$, $y(1) = 1$;
			\part $\frac{dx}{dt} = t + e^x$, $x(1) = 0$.
		\end{parts}
		
		\question
		(Filippov 225)
		Show the region of x-y plane where specified differential equation has a unique solution.
		\begin{parts}
			\part $y' = 2xy + y^2$;
			\part $(x - 2)y' = \sqrt{y} -x$;
			\part $(y - x)y' = y\ln x$;
			\part $y' = 2 + \sqrt[3]{y - 2x}$;
			\part $y' = 1 + \tan y$.
		\end{parts}
	\end{questions}
